Ad Campaign Pacing Tool

document.addEventListener("DOMContentLoaded", function () { const form = document.getElementById("campaign-form"); const result = document.getElementById("result"); form.addEventListener("submit", function (e) { e.preventDefault(); const campaignName = document.getElementById("campaign-name").value; const totalBudget = parseFloat(document.getElementById("total-budget").value); const desiredImpressions = parseFloat(document.getElementById("desired-impressions").value); const cpm = parseFloat(document.getElementById("cpm").value); const startDate = new Date(document.getElementById("start-date").value); const endDate = new Date(document.getElementById("end-date").value); const totalDays = (endDate - startDate) / (1000 * 60 * 60 * 24) + 1; const dailyBudget = (totalBudget / totalDays).toFixed(2); const dailyImpressions = (desiredImpressions * 1000 / totalDays).toFixed(2); result.innerHTML = `Campaign Name: ${campaignName}
Daily Budget: $${dailyBudget}
Daily Impressions: ${dailyImpressions}`; result.style.display = "block"; }); });
